365  template<typename PostsynapticModel>
366  SynapseGroup *addSlaveSynapsePopulation(const std::string &name, const std::string &weightSharingMasterName, unsigned int delaySteps, const std::string &src, const std::string &trg,
367  const PostsynapticModel *psm, const typename PostsynapticModel::ParamValues &postsynapticParamValues, const typename PostsynapticModel::VarValues &postsynapticVarInitialisers)
368  {
369  // Get source and target neuron groups
370  auto srcNeuronGrp = findNeuronGroupInternal(src);
371  auto trgNeuronGrp = findNeuronGroupInternal(trg);
372 
373  // Find weight sharing master group
374  auto masterGrp = findSynapseGroupInternal(weightSharingMasterName);
375  const auto *wum = masterGrp->getWUModel();
376 
377  // If the weight sharing master has individuak weights and any are read-write, give error
378  const auto wumVars = wum->getVars();
379  if((masterGrp->getMatrixType() & SynapseMatrixWeight::INDIVIDUAL) &&
380  std::any_of(wumVars.cbegin(), wumVars.cend(),
381  [](const Models::Base::Var &v)
382  {
383  return (v.access == VarAccess::READ_WRITE);
384  }))
385  {
386  throw std::runtime_error("Individual synapse variables can only be shared if they are read-only");
387  }
388 
389  // Check that population sizes match
390  if ((srcNeuronGrp->getNumNeurons() != masterGrp->getSrcNeuronGroup()->getNumNeurons())
391  || (trgNeuronGrp->getNumNeurons() != masterGrp->getTrgNeuronGroup()->getNumNeurons()))
392  {
393  throw std::runtime_error("Size of populations connected by shared weights must match");
394  }
395 
396  // If weight update model has any pre or postsynaptic variables, give error
397  // **THINK** this could be supported but quite what the semantics are is ambiguous
398  if(!wum->getPreVars().empty() || !wum->getPostVars().empty()) {
399  throw std::runtime_error("Synapse groups with pre and postsynpatic variables cannot be shared");
400  }
401 
402  // Add synapse group to map
403  auto result = m_LocalSynapseGroups.emplace(
404  std::piecewise_construct,
405  std::forward_as_tuple(name),
406  std::forward_as_tuple(name, masterGrp, masterGrp->getMatrixType(), delaySteps,
407  wum, masterGrp->getWUParams(), masterGrp->getWUVarInitialisers(), masterGrp->getWUPreVarInitialisers(), masterGrp->getWUPostVarInitialisers(),
408  psm, postsynapticParamValues.getValues(), postsynapticVarInitialisers.getInitialisers(),
409  srcNeuronGrp, trgNeuronGrp, masterGrp->getConnectivityInitialiser(),
410  m_DefaultVarLocation, m_DefaultExtraGlobalParamLocation, m_DefaultSparseConnectivityLocation, m_DefaultNarrowSparseIndEnabled));
411 
412  if(!result.second) {
413  throw std::runtime_error("Cannot add a synapse population with duplicate name:" + name);
414  }
415  else {
416  return &result.first->second;
417  }
418  }
