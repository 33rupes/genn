157 void runexpHH(float t)
158 {
159  // calculate membrane potential
160  double Imem;
161  unsigned int mt;
162  double mdt= DT/100.0;
163  for (mt=0; mt < 100; mt++) {
164  IsynGHH= 200.0*(stepVGHH-Vexp);
165  // cerr << IsynGHH << " " << Vexp << endl;
166  Imem= -(mexp*mexp*mexp*hexp*gNaexp*(Vexp-(ENaexp))+
167  nexp*nexp*nexp*nexp*gKexp*(Vexp-(EKexp))+
168  glexp*(Vexp-(Elexp))-IsynGHH);
169  double _a= (3.5+0.1*Vexp) / (1.0-exp(-3.5-0.1*Vexp));
170  double _b= 4.0*exp(-(Vexp+60.0)/18.0);
171  mexp+= (_a*(1.0-mexp)-_b*mexp)*mdt;
172  _a= 0.07*exp(-Vexp/20.0-3.0);
173  _b= 1.0 / (exp(-3.0-0.1*Vexp)+1.0);
174  hexp+= (_a*(1.0-hexp)-_b*hexp)*mdt;
175  _a= (-0.5-0.01*Vexp) / (exp(-5.0-0.1*Vexp)-1.0);
176  _b= 0.125*exp(-(Vexp+60.0)/80.0);
177  nexp+= (_a*(1.0-nexp)-_b*nexp)*mdt;
178  Vexp+= Imem/Cexp*mdt;
179  }
180 }
