432  template<typename PostsynapticModel>
433  SynapseGroup *addSlaveSynapsePopulation(const std::string &name, const std::string &weightSharingMasterName, unsigned int delaySteps, const std::string &src, const std::string &trg,
434  const PostsynapticModel *psm, const typename PostsynapticModel::ParamValues &postsynapticParamValues, const typename PostsynapticModel::VarValues &postsynapticVarInitialisers)
435  {
436  // Get source and target neuron groups
437  auto srcNeuronGrp = findNeuronGroupInternal(src);
438  auto trgNeuronGrp = findNeuronGroupInternal(trg);
439 
440  // Find weight sharing master group
441  auto masterGrp = findSynapseGroupInternal(weightSharingMasterName);
442  const auto *wum = masterGrp->getWUModel();
443 
444  // If the weight sharing master has individuak weights and any are read-write, give error
445  const auto wumVars = wum->getVars();
446  if((masterGrp->getMatrixType() & SynapseMatrixWeight::INDIVIDUAL) &&
447  std::any_of(wumVars.cbegin(), wumVars.cend(),
448  [](const Models::Base::Var &v)
449  {
450  return (v.access == VarAccess::READ_WRITE);
451  }))
452  {
453  throw std::runtime_error("Individual synapse variables can only be shared if they are read-only");
454  }
455 
456  // Check that population sizes match
457  if ((srcNeuronGrp->getNumNeurons() != masterGrp->getSrcNeuronGroup()->getNumNeurons())
458  || (trgNeuronGrp->getNumNeurons() != masterGrp->getTrgNeuronGroup()->getNumNeurons()))
459  {
460  throw std::runtime_error("Size of populations connected by shared weights must match");
461  }
462 
463  // If weight update model has any pre or postsynaptic variables, give error
464  // **THINK** this could be supported but quite what the semantics are is ambiguous
465  if(!wum->getPreVars().empty() || !wum->getPostVars().empty()) {
466  throw std::runtime_error("Synapse groups with pre and postsynpatic variables cannot be shared");
467  }
468 
469  // Add synapse group to map
470  auto result = m_LocalSynapseGroups.emplace(
471  std::piecewise_construct,
472  std::forward_as_tuple(name),
473  std::forward_as_tuple(name, masterGrp, masterGrp->getMatrixType(), delaySteps,
474  wum, masterGrp->getWUParams(), masterGrp->getWUVarInitialisers(), masterGrp->getWUPreVarInitialisers(), masterGrp->getWUPostVarInitialisers(),
475  psm, postsynapticParamValues.getInitialisers(), postsynapticVarInitialisers.getInitialisers(),
476  srcNeuronGrp, trgNeuronGrp, masterGrp->getConnectivityInitialiser(),
477  m_DefaultVarLocation, m_DefaultExtraGlobalParamLocation, m_DefaultSparseConnectivityLocation, m_DefaultNarrowSparseIndEnabled));
478 
479  if(!result.second) {
480  throw std::runtime_error("Cannot add a synapse population with duplicate name:" + name);
481  }
482  else {
483  return &result.first->second;
484  }
485  }
