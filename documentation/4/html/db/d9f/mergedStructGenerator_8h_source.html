179  std::sort(sortedFields.begin(), sortedFields.end(),
180  [&backend](const Field &a, const Field &b)

183  });

191  for(const auto &f : sortedFields) {
192  // Add field to structure
193  definitionsInternal << std::get<0>(f) << " " << std::get<1>(f) << ";" << std::endl;
194 
195  // Add size of field to total

208  }

216
