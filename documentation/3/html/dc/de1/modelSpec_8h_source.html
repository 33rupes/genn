272  SynapseGroup *addSynapsePopulation(const string &name, SynapseMatrixType mtype, unsigned int delaySteps, const string& src, const string& trg,
273  const typename WeightUpdateModel::ParamValues &weightParamValues, const typename WeightUpdateModel::VarValues &weightVarValues,
274  const typename PostsynapticModel::ParamValues &postsynapticParamValues, const typename PostsynapticModel::VarValues &postsynapticVarValues)
275  {
276  if (!GeNNReady) {
277  gennError("You need to call initGeNN first.");
278  }
279  if (final) {
280  gennError("Trying to add a synapse population to a finalized model.");
281  }
282 
283 
284  auto srcNeuronGrp = findNeuronGroup(src);
285  auto trgNeuronGrp = findNeuronGroup(trg);
286 
287  srcNeuronGrp->checkNumDelaySlots(delaySteps);
288  if (delaySteps != NO_DELAY)
289  {
290  needSynapseDelay = true;
291  }
292 
293  // Add synapse group
294  auto result = m_SynapseGroups.insert(
295  pair<string, SynapseGroup>(
296  name, SynapseGroup(name, mtype, delaySteps,
297  WeightUpdateModel::getInstance(), weightParamValues.getValues(), weightVarValues.getValues(),
298  PostsynapticModel::getInstance(), postsynapticParamValues.getValues(), postsynapticVarValues.getValues(),
299  srcNeuronGrp, trgNeuronGrp)));
300 
301  if(!result.second)
302  {
303  gennError("Cannot add a synapse population with duplicate name:" + name);
304  return NULL;
305  }
306  else
307  {
308  // Get pointer to new synapse group
309  SynapseGroup *newSynapseGroup = &result.first->second;
310 
311  // If the weight update model requires presynaptic
312  // spike times, set flag in source neuron group
313  if (newSynapseGroup->getWUModel()->isPreSpikeTimeRequired()) {
314  srcNeuronGrp->setSpikeTimeRequired(true);
315  needSt = true;
316  }
317 
318  // If the weight update model requires postsynaptic
319  // spike times, set flag in target neuron group
320  if (newSynapseGroup->getWUModel()->isPostSpikeTimeRequired()) {
321  trgNeuronGrp->setSpikeTimeRequired(true);
322  needSt = true;
323  }
324 
325  // Add references to target and source neuron groups
326  trgNeuronGrp->addInSyn(newSynapseGroup);
327  srcNeuronGrp->addOutSyn(newSynapseGroup);
328 
329  // Return
330  return newSynapseGroup;
331  }
332  }
