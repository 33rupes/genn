290  template<typename Group, typename MergedGroup, typename D>
291  void createMergedGroupsHash(const ModelSpecInternal &model, const BackendBase &backend,
292  const std::vector<std::reference_wrapper<const Group>> &unmergedGroups,
293  std::vector<MergedGroup> &mergedGroups, D getHashDigest)
294  {
295  // Create a hash map to group together groups with the same SHA1 digest
296  std::unordered_map<boost::uuids::detail::sha1::digest_type,
297  std::vector<std::reference_wrapper<const Group>>,
298  Utils::SHA1Hash> protoMergedGroups;
299 
300  // Add unmerged groups to correct vector
301  for(const auto &g : unmergedGroups) {
302  protoMergedGroups[(g.get().*getHashDigest)()].push_back(g);
303  }
304 
305  // Reserve final merged groups vector
306  mergedGroups.reserve(protoMergedGroups.size());
307 
308  // Loop through resultant merged groups
309  size_t i = 0;
310  for(const auto &p : protoMergedGroups) {
311  // Add group to vector
312  mergedGroups.emplace_back(i, model.getPrecision(), model.getTimePrecision(), backend, p.second);
313 
314  // Loop through fields
315  for(const auto &f : mergedGroups.back().getFields()) {
316  // If field is an EGP, add record to merged EGPS
317  if(std::get<3>(f) == MergedGroup::FieldType::PointerEGP || std::get<3>(f) == MergedGroup::FieldType::ScalarEGP) {
318  // Loop through groups within newly-created merged group
319  for(size_t groupIndex = 0; groupIndex < mergedGroups.back().getGroups().size(); groupIndex++) {
320  const auto &g = mergedGroups.back().getGroups()[groupIndex];
321 
322  // Add reference to this group's variable to data structure
323  m_MergedEGPs[std::get<2>(f)(g, groupIndex)].emplace(
324  std::piecewise_construct,
325  std::forward_as_tuple(MergedGroup::name),
326  std::forward_as_tuple(i, groupIndex, std::get<0>(f), std::get<1>(f)));
327  }
328  }
329  }
330 
331  i++;
332  }
333  }
