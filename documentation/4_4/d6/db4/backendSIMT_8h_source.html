206  void genGroupMergedSearch(CodeStream &os, Substitutions &popSubs, const T &g, size_t idStart) const
207  {
208  if(g.getGroups().size() == 1) {
209  os << getPointerPrefix() << "struct Merged" << T::name << "Group" << g.getIndex() << " *group";
210  os << " = &d_merged" << T::name << "Group" << g.getIndex() << "[0]; " << std::endl;
211  os << "const unsigned int lid = id - " << idStart << ";" << std::endl;
212 
213  // Use the starting thread ID of the whole merged group as group_start_id
214  popSubs.addVarSubstitution("group_start_id", std::to_string(idStart));
215  }
216  else {
217  // Perform bisect operation to get index of merged struct
218  os << "unsigned int lo = 0;" << std::endl;
219  os << "unsigned int hi = " << g.getGroups().size() << ";" << std::endl;
220  os << "while(lo < hi)" << std::endl;
221  {
222  CodeStream::Scope b(os);
223  os << "const unsigned int mid = (lo + hi) / 2;" << std::endl;
224 
225  os << "if(id < d_merged" << T::name << "GroupStartID" << g.getIndex() << "[mid])";
226  {
227  CodeStream::Scope b(os);
228  os << "hi = mid;" << std::endl;
229  }
230  os << "else";
231  {
232  CodeStream::Scope b(os);
233  os << "lo = mid + 1;" << std::endl;
234  }
235  }
236 
237  // Use this to get reference to merged group structure
238  os << getPointerPrefix() << "struct Merged" << T::name << "Group" << g.getIndex() << " *group";
239  os << " = &d_merged" << T::name << "Group" << g.getIndex() << "[lo - 1]; " << std::endl;
240 
241  // Get group start thread ID and use as group_start_id
242  os << "const unsigned int groupStartID = d_merged" << T::name << "GroupStartID" << g.getIndex() << "[lo - 1];" << std::endl;
243  popSubs.addVarSubstitution("group_start_id", "groupStartID");
244 
245  // Use this to calculate local id within group
246  os << "const unsigned int lid = id - groupStartID;" << std::endl;
247  }
248  popSubs.addVarSubstitution("id", "lid");
249  }

251  template<typename T>
252  void genParallelGroup(CodeStream &os, const Substitutions &kernelSubs, const std::vector<T> &groups, size_t &idStart,
253  GetPaddedGroupSizeFunc<typename T::GroupInternal> getPaddedSizeFunc,
254  GroupHandler<T> handler) const
255  {
256  // Loop through groups
257  for(const auto &gMerge : groups) {
258  // Sum padded sizes of each group within merged group
259  const size_t paddedSize = std::accumulate(
260  gMerge.getGroups().cbegin(), gMerge.getGroups().cend(), size_t{0},
261  [gMerge, getPaddedSizeFunc](size_t acc, std::reference_wrapper<const typename T::GroupInternal> g)
262  {
263  return (acc + getPaddedSizeFunc(g.get()));
264  });
265 
266  os << "// merged" << gMerge.getIndex() << std::endl;
267 
268  // If this is the first group
269  if(idStart == 0) {
270  os << "if(id < " << paddedSize << ")";
271  }
272  else {
273  os << "if(id >= " << idStart << " && id < " << idStart + paddedSize << ")";
274  }
275  {
276  CodeStream::Scope b(os);
277  Substitutions popSubs(&kernelSubs);
278 
279  genGroupMergedSearch(os, popSubs, gMerge, idStart);
280 
281  handler(os, gMerge, popSubs);
