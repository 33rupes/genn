173  template<typename T>
174  std::set<EGPField> getMergedGroupFields() const
175  {

178  for(const auto &e : m_MergedEGPs) {

234  template<typename Group, typename MergedGroup, typename M>
235  void createMergedGroups(const ModelSpecInternal &model, const BackendBase &backend,
236  std::vector<std::reference_wrapper<const Group>> &unmergedGroups,
237  std::vector<MergedGroup> &mergedGroups, M canMerge)
238  {
239  // Loop through un-merged groups
240  std::vector<std::vector<std::reference_wrapper<const Group>>> protoMergedGroups;
241  while(!unmergedGroups.empty()) {
242  // Remove last group from vector
243  const Group &group = unmergedGroups.back().get();
244  unmergedGroups.pop_back();
245 
246  // Loop through existing proto-merged groups
247  bool existingMergedGroupFound = false;
248  for(auto &p : protoMergedGroups) {
249  assert(!p.empty());
250 
251  // If our group can be merged with this proto-merged group
252  if(canMerge(p.front().get(), group)) {
253  // Add group to vector
254  p.emplace_back(group);
255 
256  // Set flag and stop searching
257  existingMergedGroupFound = true;
258  break;
259  }
260  }
261 
262  // If no existing merged groups were found,
263  // create a new proto-merged group containing just this group
264  if(!existingMergedGroupFound) {
265  protoMergedGroups.emplace_back();
266  protoMergedGroups.back().emplace_back(group);
267  }
268  }
269 
270  // Reserve final merged groups vector
271  mergedGroups.reserve(protoMergedGroups.size());
272 
273  // Loop through resultant merged groups
274  for(size_t i = 0; i < protoMergedGroups.size(); i++) {
275  // Add group to vector, moving vectors of groups into data structure to avoid copying
276  mergedGroups.emplace_back(i, model.getPrecision(), model.getTimePrecision(), backend,
277  std::move(protoMergedGroups[i]));
278 
279  // Loop through fields
280  for(const auto &f : mergedGroups.back().getFields()) {
281  // If field is an EGP, add record to merged EGPS
282  if(std::get<3>(f) == MergedGroup::FieldType::PointerEGP || std::get<3>(f) == MergedGroup::FieldType::ScalarEGP) {
283  // Loop through groups within newly-created merged group
284  for(size_t groupIndex = 0; groupIndex < mergedGroups.back().getGroups().size(); groupIndex++) {
285  const auto &g = mergedGroups.back().getGroups()[groupIndex];
286 
287  // Add reference to this group's variable to data structure
288  m_MergedEGPs[std::get<2>(f)(g, groupIndex)].emplace(
289  std::piecewise_construct,
290  std::forward_as_tuple(MergedGroup::name),
291  std::forward_as_tuple(i, groupIndex, std::get<0>(f), std::get<1>(f)));
292  }
293  }
294  }
295  }
296  }
