210  NeuronGroup *addNeuronPopulation(const std::string &name, unsigned int size, const NeuronModel *model,
211  const typename NeuronModel::ParamValues &paramValues,
212  const typename NeuronModel::VarValues &varInitialisers,
213  int hostID = 0)
214  {
215 #ifdef MPI_ENABLE
216  // Determine the host ID
217  int mpiHostID = 0;
218  MPI_Comm_rank(MPI_COMM_WORLD, &mpiHostID);
219 
220  // Pick map to add group to appropriately
221  auto &groupMap = (hostID == mpiHostID) ? m_LocalNeuronGroups : m_RemoteNeuronGroups;
222 #else
223  // If MPI is disabled always add to local neuron groups and zero host id
224  auto &groupMap = m_LocalNeuronGroups;
225  hostID = 0;
226 #endif
227 
228  // Add neuron group to map
229  auto result = groupMap.emplace(std::piecewise_construct,
230  std::forward_as_tuple(name),
231  std::forward_as_tuple(name, size, model,
232  paramValues.getValues(), varInitialisers.getInitialisers(),
233  m_DefaultVarLocation, m_DefaultExtraGlobalParamLocation, hostID));
234 
235  if(!result.second) {
236  throw std::runtime_error("Cannot add a neuron population with duplicate name:" + name);
237  }
238  else {
239  return &result.first->second;
240  }
241  }
