167 void runexpHH(float t)
168 {
169  // calculate membrane potential
170  double Imem;
171  unsigned int mt;
172  double mdt= DT/100.0;
173  for (mt=0; mt < 100; mt++) {
174  IsynGHH= 200.0*(stepVGHH-Vexp);
175  // cerr << IsynGHH << " " << Vexp << endl;
176  Imem= -(mexp*mexp*mexp*hexp*gNaexp*(Vexp-(ENaexp))+
177  nexp*nexp*nexp*nexp*gKexp*(Vexp-(EKexp))+
178  glexp*(Vexp-(Elexp))-IsynGHH);
179  double _a= (3.5+0.1*Vexp) / (1.0-exp(-3.5-0.1*Vexp));
180  double _b= 4.0*exp(-(Vexp+60.0)/18.0);
181  mexp+= (_a*(1.0-mexp)-_b*mexp)*mdt;
182  _a= 0.07*exp(-Vexp/20.0-3.0);
183  _b= 1.0 / (exp(-3.0-0.1*Vexp)+1.0);
184  hexp+= (_a*(1.0-hexp)-_b*hexp)*mdt;
185  _a= (-0.5-0.01*Vexp) / (exp(-5.0-0.1*Vexp)-1.0);
186  _b= 0.125*exp(-(Vexp+60.0)/80.0);
187  nexp+= (_a*(1.0-nexp)-_b*nexp)*mdt;
188  Vexp+= Imem/Cexp*mdt;
189  }
190 }
