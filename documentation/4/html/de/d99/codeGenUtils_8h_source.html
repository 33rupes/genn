153 void writePreciseString(std::ostream &os, T value)
154 {
155  // Cache previous precision
156  const std::streamsize previousPrecision = os.precision();
157 
158  // Set scientific formatting
159  os << std::scientific;
160 
161  // Set precision to what is required to fully represent T

163 
164  // Write value to stream

169  os.unsetf(std::ios_base::floatfield);

172  // Restore previous precision

174 }
