130  return pars[0]*pars[1]+pars[0]*pars[2];

501 class pwSTDP : public dpclass
502 {
503 public:
504  double calculateDerivedParameter(int index, vector<double> pars,
505  double dt)
506  {
507  switch (index) {
508  case 0:
509  return lim0(pars, dt);
510  case 1:
511  return lim1(pars, dt);
512  case 2:
513  return slope0(pars, dt);
514  case 3:
515  return slope1(pars, dt);
516  case 4:
517  return off0(pars, dt);
518  case 5:
519  return off1(pars, dt);
520  case 6:
521  return off2(pars, dt);
522  }
523  return -1;
524  }
525 
526  double lim0(vector<double> pars, double dt) {
527  return (1/pars[5] + 1/pars[2]) * pars[1] / (2/pars[2]);
528  }
529  double lim1(vector<double> pars, double dt) {
530  return -((1/pars[4] + 1/pars[2]) * pars[1] / (2/pars[2]));
531  }
532  double slope0(vector<double> pars, double dt) {
533  return -2*pars[6]/(pars[2]*pars[1]);
534  }
535  double slope1(vector<double> pars, double dt) {
536  return -1*slope0(pars, dt);
537  }
538  double off0(vector<double> pars, double dt) {
539  return pars[6]/pars[5];
540  }
541  double off1(vector<double> pars, double dt) {
542  return pars[6]/pars[2];
543  }
544  double off2(vector<double> pars, double dt) {
545  return pars[6]/pars[4];
546  }
547 };
