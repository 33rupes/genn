412  template<typename PostsynapticModel>
413  SynapseGroup *addSlaveSynapsePopulation(const std::string &name, const std::string &weightSharingMasterName, unsigned int delaySteps, const std::string &src, const std::string &trg,
414  const PostsynapticModel *psm, const typename PostsynapticModel::ParamValues &postsynapticParamValues, const typename PostsynapticModel::VarValues &postsynapticVarInitialisers)
415  {
416  // Get source and target neuron groups
417  auto srcNeuronGrp = findNeuronGroupInternal(src);
418  auto trgNeuronGrp = findNeuronGroupInternal(trg);
419 
420  // Find weight sharing master group
421  auto masterGrp = findSynapseGroupInternal(weightSharingMasterName);
422  const auto *wum = masterGrp->getWUModel();
423 
424  // If the weight sharing master has individuak weights and any are read-write, give error
425  const auto wumVars = wum->getVars();
426  if((masterGrp->getMatrixType() & SynapseMatrixWeight::INDIVIDUAL) &&
427  std::any_of(wumVars.cbegin(), wumVars.cend(),
428  [](const Models::Base::Var &v)
429  {
430  return (v.access == VarAccess::READ_WRITE);
431  }))
432  {
433  throw std::runtime_error("Individual synapse variables can only be shared if they are read-only");
434  }
435 
436  // Check that population sizes match
437  if ((srcNeuronGrp->getNumNeurons() != masterGrp->getSrcNeuronGroup()->getNumNeurons())
438  || (trgNeuronGrp->getNumNeurons() != masterGrp->getTrgNeuronGroup()->getNumNeurons()))
439  {
440  throw std::runtime_error("Size of populations connected by shared weights must match");
441  }
442 
443  // If weight update model has any pre or postsynaptic variables, give error
444  // **THINK** this could be supported but quite what the semantics are is ambiguous
445  if(!wum->getPreVars().empty() || !wum->getPostVars().empty()) {
446  throw std::runtime_error("Synapse groups with pre and postsynpatic variables cannot be shared");
447  }
448 
449  // Add synapse group to map
450  auto result = m_LocalSynapseGroups.emplace(
451  std::piecewise_construct,
452  std::forward_as_tuple(name),
453  std::forward_as_tuple(name, masterGrp, masterGrp->getMatrixType(), delaySteps,
454  wum, masterGrp->getWUParams(), masterGrp->getWUVarInitialisers(), masterGrp->getWUPreVarInitialisers(), masterGrp->getWUPostVarInitialisers(),
455  psm, postsynapticParamValues.getInitialisers(), postsynapticVarInitialisers.getInitialisers(),
456  srcNeuronGrp, trgNeuronGrp, masterGrp->getConnectivityInitialiser(),
457  m_DefaultVarLocation, m_DefaultExtraGlobalParamLocation, m_DefaultSparseConnectivityLocation, m_DefaultNarrowSparseIndEnabled));
458 
459  if(!result.second) {
460  throw std::runtime_error("Cannot add a synapse population with duplicate name:" + name);
461  }
462  else {
463  return &result.first->second;
464  }
465  }
