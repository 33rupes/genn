223  template<typename NeuronModel>
224  NeuronGroup *addNeuronPopulation(const string &name, unsigned int size, const NeuronModel *model,
225  const typename NeuronModel::ParamValues &paramValues,
226  const typename NeuronModel::VarValues &varInitialisers,
227  int hostID = 0, int deviceID = 0)
228  {
229  if (!GeNNReady) {
230  gennError("You need to call initGeNN first.");
231  }
232  if (final) {
233  gennError("Trying to add a neuron population to a finalized model.");
234  }
235 
236 #ifdef MPI_ENABLE
237  // Determine the host ID
238  int mpiHostID = 0;
239  MPI_Comm_rank(MPI_COMM_WORLD, &mpiHostID);
240 
241  // Pick map to add group to appropriately
242  auto &groupMap = (hostID == mpiHostID) ? m_LocalNeuronGroups : m_RemoteNeuronGroups;
243 #else
244  // If MPI is disabled always add to local neuron groups and zero host id
245  auto &groupMap = m_LocalNeuronGroups;
246  hostID = 0;
247 #endif
248 
249  // Add neuron group to map
250  auto result = groupMap.emplace(std::piecewise_construct,
251  std::forward_as_tuple(name),
252  std::forward_as_tuple(name, size, model,
253  paramValues.getValues(), varInitialisers.getInitialisers(), hostID, deviceID));
254 
255  if(!result.second)
256  {
257  gennError("Cannot add a neuron population with duplicate name:" + name);
258  return NULL;
259  }
260  else
261  {
262  return &result.first->second;
263  }
264  }
